11.2 用例

#! /usr/bin/perl
use DB_File;
use Fcntl;

$pslno="R121212121";

# 設定資料庫名稱
$database="newdata";

# 開啟資料庫 newdata，將它和 %ACLIST 這個雜湊結合在一起，只要操作  %ACLIST 等於是操作 $database 這個資料庫
tie %ACLIST, DB_File , $database, O_RDWR|O_CREAT, 0666 || die "無法開啟 DBM 資料庫 $database，原因：$!\n";

# 取出資料庫中的資料
print $ACLIST{$pslno}, "\n\n" if defined $ACLIST{$pslno}; 

# 關閉資料庫
untie %ACLIST;

# 輸出結果：
